   ```swift
   Button {
       if menuSelectionPastes {
           Task {
               await pasteService.paste(
                   content: clipping.content,
                   into: appTracker.previousApp
               )
           }
       } else {
           // Copy to clipboard without pasting
           NSPasteboard.general.clearContents()
           NSPasteboard.general.setString(clipping.content, forType: .string)
       }
   } label: {
       Text(clipping.content.prefix(displayLen))
           .lineLimit(1)
   }
   ```

  <files>FlycutSwift/App/FlycutApp.swift, FlycutSwift/Views/Settings/GeneralSettingsTab.swift</files>
  <action>
1. **FlycutApp.swift** — Make the MenuBarExtra label dynamic based on `menuIcon` setting and pause state.

   Add @AppStorage at the struct level:
   ```swift
   @AppStorage(AppSettingsKeys.menuIcon) private var menuIcon: Int = 0
   ```

   The MenuBarExtra label needs to be a computed icon name. Replace the hardcoded `systemImage:` with a dynamic label:
   ```swift
   MenuBarExtra("Flycut", systemImage: menuBarIconName) {
       MenuBarView()
           // ... existing environment injections
   }
   .menuBarExtraStyle(.menu)
   ```

   Add computed property:
   ```swift
   private var menuBarIconName: String {
       // When paused, show a distinct icon regardless of icon choice
       // Note: appDelegate.clipboardMonitor.isPaused cannot be directly observed here
       // because FlycutApp is not @Observable. Use a simpler approach:
       switch menuIcon {
       case 1: return "scissors"
       case 2: return "scissors.circle"
       case 3: return "doc.on.doc"
       default: return "doc.on.clipboard"
       }
   }
   ```

   IMPORTANT: The `.menu` style MenuBarExtra takes a `systemImage:` String parameter that is read ONCE at creation and does not dynamically update from @AppStorage changes. If dynamic icon switching does not work with `systemImage:`, use the `label:` closure variant instead:
   ```swift
   MenuBarExtra {
       MenuBarView()
           // ... environments
   } label: {
       Image(systemName: menuBarIconName)
   }
   ```

   If `MenuBarExtra` with a label closure is not available in the `.menu` style, document this limitation and the icon will update on next app launch.

2. **GeneralSettingsTab.swift** — Add icon picker in a new "Menu Bar" section:
   ```swift
   @AppStorage(AppSettingsKeys.menuIcon) private var menuIcon: Int = 0
   @AppStorage(AppSettingsKeys.menuSelectionPastes) private var menuSelectionPastes: Bool = true

   Section("Menu Bar") {
       Picker("Menu bar icon", selection: $menuIcon) {
           Label("Clipboard", systemImage: "doc.on.clipboard").tag(0)
           Label("Scissors", systemImage: "scissors").tag(1)
           Label("Scissors Circle", systemImage: "scissors.circle").tag(2)
           Label("Documents", systemImage: "doc.on.doc").tag(3)
       }
       .pickerStyle(.inline)

       Toggle("Clicking menu item pastes (vs. copy only)", isOn: $menuSelectionPastes)
   }
   ```
