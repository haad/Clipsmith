4. **BezelViewModel.swift** — Update `navigateUp()` and `navigateDown()` for wraparound:
   ```swift
   func navigateUp() {
       if UserDefaults.standard.bool(forKey: AppSettingsKeys.wraparoundBezel) {
           selectedIndex = selectedIndex > 0
               ? selectedIndex - 1
               : max(0, filteredClippings.count - 1)
       } else {
           selectedIndex = max(0, selectedIndex - 1)
       }
   }

   func navigateDown() {
       let last = max(0, filteredClippings.count - 1)
       if UserDefaults.standard.bool(forKey: AppSettingsKeys.wraparoundBezel) {
           selectedIndex = selectedIndex < last ? selectedIndex + 1 : 0
       } else {
           selectedIndex = min(last, selectedIndex + 1)
       }
   }
   ```
