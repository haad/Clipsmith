1. **ClipboardStore.swift** — Replace the current dedup logic in `insert()`:

   Current (lines 18-26):
   ```swift
   let duplicate = try modelContext.fetchCount(...)
   guard duplicate == 0 else {
       logger.debug("Duplicate clipping skipped")
       return
   }
   ```

   New:
   ```swift
   let existing = try modelContext.fetch(
       FetchDescriptor<FlycutSchemaV1.Clipping>(
           predicate: #Predicate { $0.content == content }
       )
   )
   if let duplicate = existing.first {
       // Move to top: update timestamp + source metadata
       duplicate.timestamp = .now
       duplicate.sourceAppName = sourceAppName
       duplicate.sourceAppBundleURL = sourceAppBundleURL
       try modelContext.save()
       logger.debug("Duplicate clipping moved to top")
       return
   }
   ```
