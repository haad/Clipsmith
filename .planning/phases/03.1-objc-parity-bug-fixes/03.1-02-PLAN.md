1. **BezelController.swift** — Add hold state and flags monitor:
   ```swift
   /// Track whether the bezel was opened via hotkey hold (not sticky mode).
   /// When true, releasing all modifier keys triggers paste-and-hide.
   var isHotkeyHold = false
   private var flagsMonitor: Any?
   ```

   Add flags monitor registration method (follow the EXACT same pattern as the existing `registerClickOutsideMonitor`/`removeClickOutsideMonitor` — see RESEARCH.md Pattern 3):
   ```swift
   private func registerFlagsMonitor() {
       removeFlagsMonitor()
       flagsMonitor = NSEvent.addGlobalMonitorForEvents(matching: .flagsChanged) { [weak self] event in
           guard let self, self.isHotkeyHold else { return }
           let modifiers: NSEvent.ModifierFlags = [.command, .option, .control, .shift]
           if event.modifierFlags.intersection(modifiers).isEmpty {
               Task { @MainActor in
                   await self.pasteAndHide()
                   self.isHotkeyHold = false
               }
           }
       }
   }

   private func removeFlagsMonitor() {
       if let m = flagsMonitor { NSEvent.removeMonitor(m); flagsMonitor = nil }
   }
   ```

   Do NOT use `flagsChanged(with:)` override on the panel — non-activating panels do not reliably receive these events (Pitfall 1 from RESEARCH.md). Use ONLY the global monitor approach.

   Update `show()` to register the flags monitor:
   ```swift
   func show() {
       centerOnMouseScreen()
       makeKeyAndOrderFront(nil)
       registerClickOutsideMonitor()
       if isHotkeyHold { registerFlagsMonitor() }
       logger.info("BezelController shown")
   }
   ```

   Update `hide()` to reset hold state and remove flags monitor (CRITICAL — Pitfall 7 from RESEARCH.md):
   ```swift
   func hide() {
       orderOut(nil)
       removeClickOutsideMonitor()
       removeFlagsMonitor()
       viewModel.selectedIndex = 0
       viewModel.searchText = ""
       isHotkeyHold = false
       logger.info("BezelController hidden")
   }
   ```

   Change `pasteAndHide()` from `private` to `internal` (remove `private` keyword) — needed for Wave 3 Bug #13 (double-click paste) and closure callbacks.

2. **AppDelegate.swift** — Update the activateBezel hotkey handler (lines 80-91):
   ```swift
   KeyboardShortcuts.onKeyDown(for: .activateBezel) { [weak self] in
       Task { @MainActor in
           guard let self else { return }
           let stickyBezel = UserDefaults.standard.bool(forKey: AppSettingsKeys.stickyBezel)
           if self.bezelController.isVisible {
               // Already showing — navigate down (hotkey repress)
               self.bezelController.viewModel.navigateDown()
   ```
