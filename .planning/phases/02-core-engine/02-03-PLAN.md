  <name>Task 2: Update MenuBarView with real clipping list, per-item delete, and clear-all</name>
  <files>
    FlycutSwift/Views/MenuBarView.swift
  </files>
  <action>
    **Replace the placeholder MenuBarView** with a real clipping list driven by `@Query`:

    ```swift
    import SwiftUI
    import SwiftData

    struct MenuBarView: View {
        @Query(sort: \FlycutSchemaV1.Clipping.timestamp, order: .reverse)
        private var clippings: [FlycutSchemaV1.Clipping]

        @Environment(\.modelContext) private var modelContext
        @Environment(PasteService.self) private var pasteService
        @Environment(AppTracker.self) private var appTracker

        @AppStorage(AppSettingsKeys.displayNum) private var displayNum: Int = 10
        @AppStorage(AppSettingsKeys.displayLen) private var displayLen: Int = 40

        var body: some View {
            // Show clippings up to displayNum limit
            if clippings.isEmpty {
                Text("No clippings yet")
                    .foregroundStyle(.secondary)
            } else {
                ForEach(clippings.prefix(displayNum)) { clipping in
                    Button {
                        Task {
                            await pasteService.paste(
                                content: clipping.content,
                                into: appTracker.previousApp
                            )
                        }
                    } label: {
                        Text(clipping.content.prefix(displayLen))
                            .lineLimit(1)
                    }
                    .contextMenu {
                        Button("Delete", role: .destructive) {
                            modelContext.delete(clipping)
                            try? modelContext.save()
                        }
                    }
                }
            }

            Divider()

            if !clippings.isEmpty {
                Button("Clear All") {
                    // Delete all Clipping records from the main context.
                    // Using main context here is acceptable for user-initiated delete actions.
                    for clipping in clippings {
                        modelContext.delete(clipping)
                    }
                    try? modelContext.save()
                }
            }

            Button("Preferences...") {
                NSApp.sendAction(Selector(("showSettingsWindow:")), to: nil, from: nil)
            }
            .keyboardShortcut(",", modifiers: .command)

            Button("Quit") {
                NSApplication.shared.terminate(nil)
            }
            .keyboardShortcut("q", modifiers: .command)
        }
    }
    ```

    Key design decisions:
    - `@Query` drives the list directly from SwiftData — auto-refreshes when ClipboardStore inserts on the background actor (RESEARCH.md confirms inserts trigger @Query refresh if `modelContext.save()` is called).
    - `displayNum` and `displayLen` from `@AppStorage` control visible count and truncation.
    - Paste action calls `pasteService.paste(content:into:)` with `appTracker.previousApp`.
    - **Per-item delete (CLIP-08):** Each clipping item has a `.contextMenu` with a "Delete" button (role: .destructive). Right-clicking a clipping shows the context menu; clicking Delete removes that single clipping from SwiftData and saves immediately. This is ~4 lines of SwiftUI and fully satisfies CLIP-08.
    - "Clear All" uses main context delete (acceptable for user-initiated bulk operations — not a hot path).
    - "Preferences..." opens the Settings window using `showSettingsWindow:` action.
    - NOTE: `FlycutSchemaV1.Clipping` must conform to `Identifiable` for `ForEach` to work. SwiftData `@Model` types conform to `Identifiable` via their `persistentModelID` — this is automatic, no manual conformance needed.
