    ```swift
    struct GistHistoryView: View {
        @Query(sort: \FlycutSchemaV1.GistRecord.createdAt, order: .reverse)
        private var gistHistory: [FlycutSchemaV1.GistRecord]

        @Environment(GistService.self) private var gistService

        var body: some View {
            if gistHistory.isEmpty {
                ContentUnavailableView("No Gists", systemImage: "link.badge.plus",
                    description: Text("Share a snippet or clipping as a Gist to see it here."))
            } else {
                List(gistHistory) { record in
                    HStack {
                        VStack(alignment: .leading) {
                            Text(record.filename).font(.headline)
                            Text(record.createdAt, style: .date).font(.caption).foregroundStyle(.secondary)
                        }
                        Spacer()
                        Button("Open") {
                            if let url = URL(string: record.gistURL) {
                                NSWorkspace.shared.open(url)
                            }
                        }
                    }
                    .contextMenu {
                        Button("Open in Browser") {
                            if let url = URL(string: record.gistURL) {
                                NSWorkspace.shared.open(url)
                            }
                        }
                        Button("Copy URL") {
                            NSPasteboard.general.clearContents()
                            NSPasteboard.general.setString(record.gistURL, forType: .string)
                        }
                        Divider()
                        Button("Delete Gist", role: .destructive) {
                            Task {
                                try? await gistService.deleteGist(id: record.persistentModelID)
                            }
                        }
                    }
                }
            }
        }
    }
    ```

    **Step 2: Create GistSettingsSection.swift.**

    Per user decision: "Public/private default is configurable in Settings (Gist settings section)."

    ```swift
    struct GistSettingsSection: View {
        @State private var tokenInput: String = ""
        @State private var hasToken: Bool = false
        @AppStorage("gistDefaultPublic") private var gistDefaultPublic: Bool = false

        private let tokenStore = TokenStore()

        var body: some View {
            Form {
                Section("GitHub Gist") {
                    if hasToken {
                        HStack {
                            Label("Token saved", systemImage: "checkmark.circle.fill")
                                .foregroundStyle(.green)
                            Spacer()
                            Button("Clear Token") {
                                tokenStore.deleteToken()
                                hasToken = false
                            }
                        }
                    } else {
                        SecureField("Personal Access Token", text: $tokenInput)
                        Button("Save Token") {
                            tokenStore.saveToken(tokenInput)
                            tokenInput = ""
                            hasToken = true
                        }
                        .disabled(tokenInput.isEmpty)
                        Text("Create a token at github.com/settings/tokens with the 'gist' scope.")
                            .font(.caption)
                            .foregroundStyle(.secondary)
                    }

                    Toggle("Public gists by default", isOn: $gistDefaultPublic)
                }
            }
            .onAppear {
                hasToken = tokenStore.loadToken() != nil
            }
        }
    }
    ```

    The `shareAsGist` helper:
    ```swift
    @Environment(GistService.self) private var gistService
    @AppStorage("gistDefaultPublic") private var gistDefaultPublic: Bool = false

    private func shareAsGist(content: String, language: String?, name: String) async {
        let ext = GistService.languageExtension(for: language)
        let filename = "\(name.isEmpty ? "snippet" : name.replacingOccurrences(of: " ", with: "_")).\(ext)"
        do {
            let response = try await gistService.createGist(
                filename: filename,
                content: content,
                description: name,
                isPublic: gistDefaultPublic
            )
            // URL already copied to clipboard by GistService (GIST-04)
            sendGistNotification(url: response.htmlURL, filename: filename)
        } catch GistError.noToken {
            // Show alert: "No GitHub token configured. Add one in Settings > Gist."
            showNoTokenAlert()
        } catch {
            // Show alert with error description
            showErrorAlert(error)
        }
    }
    ```

    **Step 4: macOS system notification on Gist creation.**

    Per user decision: "macOS system notification on success: 'Gist created' with URL; click notification to open in browser."

    Use `UNUserNotificationCenter` for local notifications:
    ```swift
    import UserNotifications

    func sendGistNotification(url: String, filename: String) {
        let content = UNMutableNotificationContent()
        content.title = "Gist Created"
        content.body = filename
        content.userInfo = ["url": url]
        content.sound = nil  // Claude's discretion: no sound (per CONTEXT.md)

        let request = UNNotificationRequest(
            identifier: UUID().uuidString,
            content: content,
            trigger: nil  // Deliver immediately
        )
        UNUserNotificationCenter.current().add(request)
    }
    ```

    Set up notification delegate in AppDelegate to handle click-to-open:
    ```swift
    UNUserNotificationCenter.current().delegate = self

    // In extension AppDelegate: UNUserNotificationCenterDelegate
    func userNotificationCenter(_ center: UNUserNotificationCenter,
                                didReceive response: UNNotificationResponse) async {
        if let urlString = response.notification.request.content.userInfo["url"] as? String,
           let url = URL(string: urlString) {
            NSWorkspace.shared.open(url)
        }
    }
    ```

    Request notification authorization in applicationDidFinishLaunching:
    ```swift
    UNUserNotificationCenter.current().requestAuthorization(options: [.alert]) { _, _ in }
    ```
